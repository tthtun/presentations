- Run

drmf; \

## Demo 1

docker container run \
  --detach \
  --env 'ACCEPT_EULA=Y' \
  --env 'MSSQL_SA_PASSWORD=DockerCon!!!' \
  --publish 1433:1433 \
  --network bb-net \
  --name sql-db \
  microsoft/mssql-server-linux:2017-CU1

Connect and run SQL script
